//给map添加click事件,用于侦听单击画线 google.maps.event.addListener(map, 'click', function(event) { if (isLine) { linePath.push(event.latLng); for(var i=0;i<linePath.length;i++) { } //画多边形 if (lineArea != null) { lineArea.setMap(null); } lineArea = new google.maps.Polyline({ path: linePath, //路径数组 strokeColor: "#FF0000", strokeOpacity: 1.0, strokeWeight: 1 //粗细 // fillColor: "#FF0000",//填充色 // fillOpacity: 0.1//透明度 }); lineArea.setMap(map); } else { linePath.length = 0; if (lineArea != null) { lineArea.setMap(null); } } }); //添加双击事件 google.maps.event.addListener(map, 'dblclick', function(event) { //清空线条 if (lineArea != null) { lineArea.setMap(null); } //绘制区域图 var PolyArea=new google.maps.Polygon( { path: linePath, //路径数组 strokeColor: "#FF0000", strokeOpacity: 1.0, strokeWeight: 1, //粗细 fillColor: "#FF0000",//填充色 fillOpacity: 0.1//透明度 }); PolyArea.setMap(map); });
